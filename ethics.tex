\centering{\bf \large {منشور اخلاق پژوهش}}\\
\vspace{1.5cm}
\doublespacing
\small
\centering{\textbf {با استعانت از خدای سبحان و با اعتقاد راسخ به اینکه عالم محضر خداست و او همواره ناظر بر اعمال ماست}}\\
\centering {\textbf{ و به منظور انجام شایسته پژوهش‌های اصیل، تولید دانش جدید و بهسازی زندگانی بشر،}}\\
\centering {\textbf{ ما دانشجویان و اعضای هیأت علمی دانشگاه‌ها و پژوهشگاه‌های کشور:}}\\
\vspace{1cm}
\centering{تمام تلاش خود را برای کشف حقیقت و فقط حقیقت به کار خواهیم بست و از هرگونه جعل و تحریف در فعالیت‌های علمی پرهیز می‌کنیم.}\\
\mbox{\centering{حقوق پژوهشگران، پژوهیدگان (انسان، حیوان، گیاه و اشیاء)، سازمان‌ها و سایر صاحبان حقوق را به رسمیت می‌شناسیم و در حفظ آن می‌کوشیم.}}\\
\mbox{\centering{به مالکیت مادی و معنوی آثار پژوهشی ارج می‌نهیم، برای انجام پژوهشی اصیل، اهتمام ورزیده، از سرقت علمی و ارجاع نامناسب اجتناب می‌کنیم.}}\\
\centering{ضمن پایبندی به انصاف و اجتناب از هرگونه تبعیض و تعصب، در کلیه فعالیت‌های پژوهشی، رهیافتی نقادانه اتخاذ خواهیم کرد.}\\
\centering{ضمن امانت‌داری، از منابع و امکانات اقتصادی، انسانی و فنی موجود، استفاده بهره‌ورانه خواهیم کرد.}\\
\centering{از انتشار غیراخلاقی نتایج پژوهش، نظیر انتشار موازی همپوشان و چندگانه (تکه‌ای) پرهیز می‌کنیم.}\\
\centering{اصل محرمانه بودن و رازداری را محور تمام فعالیت‌های پژوهشی خود قرار می‌دهیم.}\\
\centering{در همه فعالیت‌های پژوهشی به منافع ملی توجه کرده و برای تحقق آن می‌کوشیم.}\\
\vspace{1.5cm}
\centering{خویش را ملزم به رعایت کلیه هنجارهای علمی رشته خود، قوانین و مقررات، سیاست‌های حرفه‌ای، سازمانی، دولتی و راهبردی ملی}\\
\centering{در همه مراحل پژوهش می‌دانیم.}\\
\centering{رعایت اصول اخلاق در پژوهش را اقدامی فرهنگی می‌دانیم و به منظور بالندگی این فرهنگ، به ترویج و اشاعه آن}\\
\centering{در جامعه اهتمام می‌ورزیم.}\\
